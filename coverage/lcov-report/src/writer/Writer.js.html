const Abstract = require("../object/Abstract");
const lineSeparator = "\n";
class Writer extends Abstract {
    constructor() {
        super();
    }
 
    print(s) {
        if (s == null) {
            s = "";
        }
        this.write(s);
    }
 
    write(...args) {
        this.DT("write", args)
    }
 
 
    println(x) {
        if (x != null) {
            this.print(x);
        }
        this.write(lineSeparator);
    }
 
 
 
}
